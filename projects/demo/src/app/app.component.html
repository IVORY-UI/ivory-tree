Example: Simple Tree

@for (item of treeData; track $index) { @for (childItem of item.children; track $index) { @for (grandchildItem of childItem.children; track $index) { } } } 
          
            <ivory-tree>
              &for (item of treeData; track $index) {
                <ivory-treenode
                  [node]="item"
                  [nodeTitle]="item.name"
                  [nodeHasChildren]="item.children.length>0"
                  [nodeExpanded]="true"
                >
                  &for (childItem of item.children; track $index) {
                    <ivory-treenode
                      [node]="childItem"
                      [nodeHasChildren]="childItem.children.length>0"
                      [nodeTitle]="childItem.name"
                      [nodeExpanded]="true"
                    >
                      &for (grandchildItem of childItem.children; track $index) {
                        <ivory-treenode
                          [node]="grandchildItem"
                          [nodeTitle]="grandchildItem.name"
                        ></ivory-treenode>
                      }
                    </ivory-treenode>
                  }
                </ivory-treenode>
              }
            </ivory-tree>
